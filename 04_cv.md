<script>
  (function () {
    const resumes = {
      en: {
        html: "{{ '/resume-EN.html' | relative_url }}",
        lang: "en-CA",
        title: "Résumé — English",
        downloadLabel: "Download PDF",
        downloadShort: "PDF",
        jumpLabel: "Jump to section…",
        loading: "Loading résumé…",
        error: "Unable to load the résumé."
      },
      fr: {
        html: "{{ '/resume-FR.html' | relative_url }}",
        lang: "fr-CA",
        title: "CV — Français",
        downloadLabel: "Télécharger le PDF",
        downloadShort: "PDF",
        jumpLabel: "Aller à la section…",
        loading: "Chargement du CV…",
        error: "Impossible de charger le CV."
      }
    };

    /* The résumé files are print-first documents. These rules adapt them to
       screens without touching the print layout, and live here so they survive
       the résumé HTML being regenerated. */
    const SCREEN_CSS = `
@media screen {
  .page-container {
    box-sizing: border-box;
    min-height: 0;
    box-shadow: 0 1px 4px rgba(16, 42, 60, 0.10);
  }
  .cv section { scroll-margin-top: 64px; }
}

@media screen and (max-width: 760px) {
  .page-container { padding: 24px 20px; }
  .cv h1 { font-size: 26px !important; }
  .sec-title { align-items: flex-start; }
  .sec-title .bar { flex: 0 0 auto; margin-top: 3px; }

  /* Stack title over meta: the flex row cramps both at this width. */
  .entry-top { display: block; }
  .entry-meta { display: block; white-space: normal; margin-top: 1px; }

  /* Label/value tables read better stacked than as two narrow columns. */
  .contribtable tr,
  .skilltable tr { display: block; padding: 7px 0; }
  .contribtable td,
  .skilltable td { display: block; width: auto; padding: 0; border-bottom: 0; }
  .contribtable td.label,
  .skilltable td.label { width: auto; padding: 0 0 1px; }
  .contribtable tr { border-bottom: 0.35pt solid #CBD7DE; }

  /* Four columns will not fit a phone; render each row as a stacked card. */
  .mentortable tr:has(th) { display: none; }
  .mentortable tr {
    display: block;
    padding: 8px 0;
    border-bottom: 0.35pt solid #CBD7DE;
  }
  .mentortable tr.group-start { border-top: 0.6pt solid #CBD7DE; }
  .mentortable td { display: block; padding: 1px 0; border: 0; }
  .mentortable td:nth-child(3) { font-size: 11px; }
  .mentortable td.period { text-align: left; font-size: 10px; }
}`;

    const resumeDocument = document.getElementById('resume-document');
    const resumeRoot = resumeDocument.attachShadow({ mode: 'open' });
    const download = document.getElementById('resume-download');
    const downloadLong = download.querySelector('[data-download-long]');
    const downloadShort = download.querySelector('[data-download-short]');
    const jump = document.getElementById('resume-jump');
    const buttons = document.querySelectorAll('[data-resume-language]');
    let loadSequence = 0;

    function showMessage(text) {
      const message = document.createElement('p');
      message.style.cssText = 'padding:2rem;text-align:center';
      message.textContent = text;
      resumeRoot.replaceChildren(message);
    }

    function buildJumpMenu(page, resume) {
      const options = [new Option(resume.jumpLabel, '')];
      page.querySelectorAll('section').forEach(function (section, index) {
        const title = section.querySelector('.sec-title');
        if (!title) {
          return;
        }
        section.id = 'resume-section-' + index;
        options.push(new Option(title.textContent.trim(), section.id));
      });
      jump.replaceChildren(...options);
      jump.disabled = options.length < 2;
    }

    async function selectLanguage(language) {
      const resume = resumes[language];
      const sequence = ++loadSequence;

      resumeDocument.setAttribute('aria-label', resume.title);
      resumeDocument.lang = resume.lang;
      downloadLong.textContent = resume.downloadLabel;
      downloadShort.textContent = resume.downloadShort;
      download.setAttribute('aria-label', resume.downloadLabel);
      download.disabled = true;
      jump.disabled = true;
      jump.replaceChildren(new Option(resume.jumpLabel, ''));
      buttons.forEach(function (button) {
        button.setAttribute('aria-pressed', button.dataset.resumeLanguage === language);
      });
      history.replaceState(null, '', language === 'fr' ? '#fr' : '#en');

      showMessage(resume.loading);

      try {
        const response = await fetch(resume.html, { cache: 'no-store' });
        if (!response.ok) {
          throw new Error('HTTP ' + response.status);
        }

        const source = await response.text();
        if (sequence !== loadSequence) {
          return;
        }

        const parsed = new DOMParser().parseFromString(source, 'text/html');
        const page = parsed.querySelector('.page-container');
        if (!page) {
          throw new Error('Résumé content was not found');
        }

        const hostStyle = document.createElement('style');
        hostStyle.textContent = ':host{display:block;font-family:"Helvetica Neue",Helvetica,Arial,sans-serif;color:#202B33;font-size:12px;line-height:1.6}';
        const styles = Array.from(parsed.querySelectorAll('style')).map(function (sourceStyle) {
          const style = document.createElement('style');
          style.textContent = sourceStyle.textContent;
          return style;
        });
        const screenStyle = document.createElement('style');
        screenStyle.textContent = SCREEN_CSS;

        const pageNode = page.cloneNode(true);
        resumeRoot.replaceChildren(hostStyle, ...styles, screenStyle, pageNode);
        buildJumpMenu(pageNode, resume);
        download.disabled = false;
      } catch (error) {
        if (sequence === loadSequence) {
          showMessage(resume.error);
          console.error('Unable to load résumé:', error);
        }
      }
    }

    buttons.forEach(function (button) {
      button.addEventListener('click', function () {
        selectLanguage(button.dataset.resumeLanguage);
      });
    });

    const reduceMotion = window.matchMedia('(prefers-reduced-motion: reduce)');

    jump.addEventListener('change', function () {
      const target = jump.value && resumeRoot.getElementById(jump.value);
      if (target) {
        target.scrollIntoView({
          behavior: reduceMotion.matches ? 'auto' : 'smooth',
          block: 'start'
        });
      }
      jump.selectedIndex = 0;
    });

    download.addEventListener('click', function () {
      window.print();
    });

    const hashLanguage = window.location.hash.toLowerCase();
    const selectedButton = document.querySelector('[data-resume-language][aria-pressed="true"]');
    const initialLanguage = hashLanguage === '#fr' || hashLanguage === '#en'
      ? hashLanguage.slice(1)
      : selectedButton.dataset.resumeLanguage;
    selectLanguage(initialLanguage);
  }());
</script>
